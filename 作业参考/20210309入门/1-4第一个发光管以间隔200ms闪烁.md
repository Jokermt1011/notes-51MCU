#第一个发光管以间隔200ms闪烁
##并行控制
```c
#include <reg51.h>
#include <intrins.h>
#define uint unsigned int

void delayms(uint x){
	uint i,j;
	for(i=x;i>0;i--)
		for(j=110;j>0;j--); //不精准的延迟，可替换_nop_函数
}

void main(void){
	while(1){
		P1=0xfe;
		delayms(200);
		P1=0xff;
		delayms(200);
	}
}
```
##设定单个
```c
#include <reg51.h>
#include <intrins.h>
#define uint unsigned int

sbit led1=P1^0;

void delayms(uint x){
	uint i,j;
	for(i=x;i>0;i--)
		for(j=110;j>0;j--); //不精准的延迟，可替换_nop_函数
}

void main(void){
	P1=0xff;    //初始化
	while(1){
		led1=0;
		delayms(200);
		led1=1;
		delayms(200);
	}
}
```
